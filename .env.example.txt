# =========================================
# üöÄ App Configuration
# =========================================
# Environment mode for the Node.js server (development or production)
NODE_ENV=development

# Environment mode for the Python/Flask server (development or production)
# In development, this enables debug mode.
PY_ENV=development

# Default port for the Python API server
PORT=8080


# =========================================
# üîê Authentication Secrets
# =========================================
# A long, random string used to sign session cookies for the Node.js auth server.
# Generate one here: https://www.uuidgenerator.net/
SESSION_SECRET=your_super_secret_session_key_here

# A long, random string used to sign JSON Web Tokens (JWTs).
# This MUST be the same for both the Node.js auth server and the Python server.
JWT_SECRET=your_super_secret_jwt_key_here


# =========================================
# ü§ñ AI Integrations
# =========================================
# Your Google Gemini API key, used by the Python server.
# Get one from Google AI Studio: https://aistudio.google.com/
